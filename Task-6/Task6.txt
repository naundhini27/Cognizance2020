BEGIN
DISPLAY 'Enter the number'
INPUT user inputs the number
STORE the users input in the variable no
SET sum=0
SET original=no
WHILE no>0 
	CALCULATE rem=no MOD 10
	CALCULATE sum=sum + (rem)^3
	CALCULATE no=no/10
END WHILE
IF sum=original
   THEN
	DISPLAY The number is not an Armstrong number
   ELSE
	DISPLAY The number is an Armstrong number
END
